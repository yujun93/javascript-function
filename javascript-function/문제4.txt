    function print_star(n) {

      var result = '';



      for(var i=0; i<n; i++){

        for(var j=0; j<(n-i-1); j++){

          result += ' ';

        }

        for(var j=n; j>(n-i-1); j--){

          result += '*';

        }

        result += '\n';

      }

      return result;

    }



    exepect(print_star(1), '*\n');

    exepect(print_star(2), ' *\n**\n');

    exepect(print_star(3), '  *\n **\n***\n');

    exepect(print_star(4), '   *\n  **\n ***\n****\n');

    exepect(print_star(5), '    *\n   **\n  ***\n ****\n*****\n');
